import React, { useState } from 'react';
import { motion, AnimatePresence } from 'motion/react';
import { 
  Building2, 
  User, 
  Phone, 
  Mail, 
  MapPin, 
  Stethoscope, 
  FileText, 
  Banknote, 
  Percent, 
  CheckCircle2, 
  AlertCircle,
  Loader2,
  LogIn
} from 'lucide-react';
import { auth, db, handleFirestoreError, OperationType, signInWithGoogle } from '../lib/firebase';
import { collection, doc, setDoc, serverTimestamp } from 'firebase/firestore';
import { blackBoxFilter, getPostingDateTime } from '../lib/security';

export const ReferralForm = () => {
  const [isLoading, setIsLoading] = useState(false);
  const [isSuccess, setIsSuccess] = useState(false);
  const [user, setUser] = useState(auth.currentUser);
  const [error, setError] = useState<string | null>(null);

  const [formData, setFormData] = useState({
    establishmentName: '',
    managerName: '',
    phone: '',
    email: '',
    location: '',
    specialization: '',
    acceptanceProcedures: '',
    residentSalary: '',
    subordinatePosition: '',
    subordinateSalary: '',
    subordinatePercentage: '',
  });

  React.useEffect(() => {
    const unsubscribe = auth.onAuthStateChanged((u) => {
      setUser(u);
      if (u) {
        setFormData(prev => ({...prev, email: prev.email || u.email || ''}));
      }
    });
    return () => unsubscribe();
  }, []);

  const handleSubmit = async (e: React.FormEvent) => {
    e.preventDefault();
    if (!user) {
      const u = await signInWithGoogle();
      if (!u) return;
    }

    setIsLoading(true);
    setError(null);
    try {
      const referralRef = doc(collection(db, 'referrals'));
      const payload = {
        ...formData,
        createdAt: serverTimestamp(),
        post_date: getPostingDateTime(),
        userId: auth.currentUser?.uid,
      };

      await setDoc(referralRef, payload);
      setIsSuccess(true);
    } catch (err: any) {
      setError("حدث خطأ أثناء إرسال الإحالة. يرجى المحاولة مرة أخرى.");
      handleFirestoreError(err, OperationType.CREATE, 'referrals');
    } finally {
      setIsLoading(false);
    }
  };

  if (isSuccess) {
    return (
      <motion.div 
        initial={{ opacity: 0, scale: 0.95 }}
        animate={{ opacity: 1, scale: 1 }}
        className="text-center py-12 px-6 bg-green-50 rounded-3xl border border-green-100"
      >
        <div className="w-16 h-16 bg-green-100 text-green-600 rounded-full flex items-center justify-center mx-auto mb-6">
          <CheckCircle2 className="w-10 h-10" />
        </div>
        <h2 className="text-2xl font-bold text-green-900 mb-2">تم نشر الإحالة بنجاح</h2>
        <p className="text-green-700 mb-8">سيتم عرض طلب التوظيف الخاص بكم للمختصين المسجلين لدينا.</p>
        <button 
          onClick={() => { setIsSuccess(false); setFormData({
            establishmentName: '', managerName: '', phone: '', email: '', location: '', 
            specialization: '', acceptanceProcedures: '', residentSalary: '',
            subordinatePosition: '', subordinateSalary: '', subordinatePercentage: ''
          }); }}
          className="bg-green-600 text-white px-8 py-3 rounded-xl font-bold hover:bg-green-700 transition-colors"
        >
          نشر إحالة أخرى
        </button>
      </motion.div>
    );
  }

  return (
    <div className="space-y-8">
      {!user && (
        <div className="p-4 bg-amber-50 border border-amber-100 rounded-2xl flex items-center justify-between gap-4">
          <p className="text-sm font-medium text-amber-800">يرجى تسجيل الدخول لنشر إحالة طبية.</p>
          <button 
            onClick={signInWithGoogle}
            className="px-4 py-2 bg-white text-slate-800 rounded-xl font-bold border border-slate-200 flex items-center gap-2"
          >
            <LogIn className="w-4 h-4" /> دخول
          </button>
        </div>
      )}

      {error && (
        <div className="p-4 bg-red-50 text-red-600 rounded-xl flex items-center gap-2 text-sm font-bold border border-red-100">
          <AlertCircle className="w-5 h-5" /> {error}
        </div>
      )}

      <form onSubmit={handleSubmit} className="space-y-8">
        <div className="space-y-6">
          <h3 className="text-lg font-bold text-[#002366] flex items-center gap-2 border-r-4 border-[#D4AF37] pr-3">بيانات المنشأة</h3>
          <div className="grid md:grid-cols-2 gap-6">
            <div className="space-y-2">
              <label className="text-sm font-bold text-slate-700">اسم المنشأة الطبية</label>
              <div className="relative">
                <input 
                  type="text" 
                  value={formData.establishmentName}
                  onChange={(e) => setFormData({...formData, establishmentName: blackBoxFilter(e.target.value)})}
                  className="w-full bg-white border border-slate-200 rounded-xl px-10 py-3 focus:ring-2 focus:ring-[#D4AF37] outline-none" 
                  placeholder="مستشفى / مركز / عيادة..."
                  required 
                />
                <Building2 className="absolute right-3 top-3.5 w-5 h-5 text-slate-400" />
              </div>
            </div>
            <div className="space-y-2">
              <label className="text-sm font-bold text-slate-700">اسم المدير المسئول</label>
              <div className="relative">
                <input 
                  type="text" 
                  value={formData.managerName}
                  onChange={(e) => setFormData({...formData, managerName: blackBoxFilter(e.target.value)})}
                  className="w-full bg-white border border-slate-200 rounded-xl px-10 py-3 focus:ring-2 focus:ring-[#D4AF37] outline-none" 
                  required 
                />
                <User className="absolute right-3 top-3.5 w-5 h-5 text-slate-400" />
              </div>
            </div>
            <div className="space-y-2">
              <label className="text-sm font-bold text-slate-700">رقم الهاتف للتواصل</label>
              <div className="relative">
                <input 
                  type="tel" 
                  value={formData.phone}
                  onChange={(e) => setFormData({...formData, phone: e.target.value})}
                  className="w-full bg-white border border-slate-200 rounded-xl px-10 py-3 focus:ring-2 focus:ring-[#D4AF37] outline-none" 
                  required 
                />
                <Phone className="absolute right-3 top-3.5 w-5 h-5 text-slate-400" />
              </div>
            </div>
            <div className="space-y-2">
              <label className="text-sm font-bold text-slate-700">البريد الإلكتروني</label>
              <div className="relative">
                <input 
                  type="email" 
                  value={formData.email}
                  onChange={(e) => setFormData({...formData, email: e.target.value})}
                  className="w-full bg-white border border-slate-200 rounded-xl px-10 py-3 focus:ring-2 focus:ring-[#D4AF37] outline-none" 
                  required 
                />
                <Mail className="absolute right-3 top-3.5 w-5 h-5 text-slate-400" />
              </div>
            </div>
          </div>
          <div className="space-y-2">
            <label className="text-sm font-bold text-slate-700">الموقع الدقيق</label>
            <div className="relative">
              <input 
                type="text" 
                value={formData.location}
                onChange={(e) => setFormData({...formData, location: blackBoxFilter(e.target.value)})}
                className="w-full bg-white border border-slate-200 rounded-xl px-10 py-3 focus:ring-2 focus:ring-[#D4AF37] outline-none" 
                placeholder="المدينة، المنطقة، الشارع..."
                required 
              />
              <MapPin className="absolute right-3 top-3.5 w-5 h-5 text-slate-400" />
            </div>
          </div>
        </div>

        <div className="space-y-6">
          <h3 className="text-lg font-bold text-[#002366] flex items-center gap-2 border-r-4 border-[#D4AF37] pr-3">متطلبات الإحالة</h3>
          <div className="space-y-6">
             <div className="grid md:grid-cols-2 gap-6">
                <div className="space-y-2">
                  <label className="text-sm font-bold text-slate-700">التخصص المطلوب</label>
                  <div className="relative">
                    <input 
                      type="text" 
                      value={formData.specialization}
                      onChange={(e) => setFormData({...formData, specialization: blackBoxFilter(e.target.value)})}
                      className="w-full bg-white border border-slate-200 rounded-xl px-10 py-3 focus:ring-2 focus:ring-[#D4AF37] outline-none" 
                      required 
                    />
                    <Stethoscope className="absolute right-3 top-3.5 w-5 h-5 text-slate-400" />
                  </div>
                </div>
                <div className="space-y-2">
                  <label className="text-sm font-bold text-slate-700">الراتب للمقيم</label>
                  <div className="relative">
                    <input 
                      type="text" 
                      value={formData.residentSalary}
                      onChange={(e) => setFormData({...formData, residentSalary: e.target.value})}
                      className="w-full bg-white border border-slate-200 rounded-xl px-10 py-3 focus:ring-2 focus:ring-[#D4AF37] outline-none" 
                      placeholder="الراتب الأساسي أو المدى..."
                      required 
                    />
                    <Banknote className="absolute right-3 top-3.5 w-5 h-5 text-slate-400" />
                  </div>
                </div>
             </div>

             <div className="space-y-2">
                <label className="text-sm font-bold text-slate-700">إجراءات القبول</label>
                <div className="relative">
                  <textarea 
                    value={formData.acceptanceProcedures}
                    onChange={(e) => setFormData({...formData, acceptanceProcedures: blackBoxFilter(e.target.value)})}
                    className="w-full bg-white border border-slate-200 rounded-xl px-10 py-3 focus:ring-2 focus:ring-[#D4AF37] outline-none min-h-[100px]" 
                    placeholder="الاختبارات، المقابلات، الأوراق المطلوبة..."
                    required 
                  />
                  <FileText className="absolute right-3 top-3.5 w-5 h-5 text-slate-400" />
                </div>
             </div>

             <div className="bg-slate-100/50 p-6 rounded-2xl border border-slate-200 space-y-4">
                <h4 className="font-bold text-slate-700">بيانات التابع (اختياري)</h4>
                <div className="grid md:grid-cols-3 gap-4">
                  <div className="space-y-2">
                    <label className="text-xs font-bold text-slate-500">منصب التابع</label>
                    <input 
                      type="text" 
                      value={formData.subordinatePosition}
                      onChange={(e) => setFormData({...formData, subordinatePosition: e.target.value})}
                      className="w-full bg-white border border-slate-200 rounded-lg px-3 py-2 text-sm focus:ring-1 focus:ring-[#D4AF37] outline-none" 
                    />
                  </div>
                  <div className="space-y-2">
                    <label className="text-xs font-bold text-slate-500">راتب التابع</label>
                    <input 
                      type="text" 
                      value={formData.subordinateSalary}
                      onChange={(e) => setFormData({...formData, subordinateSalary: e.target.value})}
                      className="w-full bg-white border border-slate-200 rounded-lg px-3 py-2 text-sm focus:ring-1 focus:ring-[#D4AF37] outline-none" 
                    />
                  </div>
                  <div className="space-y-2">
                    <label className="text-xs font-bold text-slate-500">نسبة التابع</label>
                    <div className="relative">
                      <input 
                        type="text" 
                        value={formData.subordinatePercentage}
                        onChange={(e) => setFormData({...formData, subordinatePercentage: e.target.value})}
                        className="w-full bg-white border border-slate-200 rounded-lg px-3 py-2 pl-8 text-sm focus:ring-1 focus:ring-[#D4AF37] outline-none" 
                      />
                      <Percent className="absolute left-3 top-2.5 w-4 h-4 text-slate-400" />
                    </div>
                  </div>
                </div>
             </div>
          </div>
        </div>

        <button 
          type="submit"
          disabled={isLoading}
          className="w-full bg-[#002366] text-white py-4 rounded-xl font-bold hover:bg-[#001a4d] transition-all flex items-center justify-center gap-2 shadow-lg active:scale-95 disabled:opacity-50"
        >
          {isLoading ? <Loader2 className="w-5 h-5 animate-spin" /> : <CheckCircle2 className="w-5 h-5" />}
          نشر الإحالة الطبية الآن
        </button>
      </form>
    </div>
  );
};
